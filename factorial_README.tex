\section*{Description}
This shell script calculates the factorial of a non-negative integer provided by the user.

\section*{Features}
\begin{itemize}
  \item Validates the input to ensure it is a non-negative integer.
  \item Uses a \texttt{for} loop to compute the factorial.
\end{itemize}

\section*{Usage}
\begin{lstlisting}[language=bash]
chmod +x factorial.sh
./factorial.sh
\end{lstlisting}

\section*{Sample Output}
\begin{lstlisting}
Enter a number: 5
Factorial of 5 is 120
\end{lstlisting}

\section*{Requirements}
\begin{itemize}
  \item Bash shell
